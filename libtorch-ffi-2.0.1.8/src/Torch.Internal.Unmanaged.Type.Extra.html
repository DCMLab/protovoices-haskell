_obj =
  [C.throwBlock| std::vector<at::Dimname>* {
      auto ref = (*$(at::Tensor* _obj)).names();
      std::vector<at::Dimname>* vec = new std::vector<at::Dimname>();
      for(int i=0;i<ref.size();i++){
        vec->push_back(ref[i]);
      }
      return vec;
  }|]
