-- | Infix alias of 'imap'.  Or, ('<$>') for 'IxFunctor'.  Should be
-- interchangeable with ('<$>'), but requires 'IxFunctor' constraints instead
-- of (possibly many) 'Functor' constraints.

<<$>> :: (a -> b) -> f j k a -> f j k b
(<<$>>) = (a -> b) -> f j k a -> f j k b
forall k k (f :: k -> k -> * -> *) a b (j :: k) (k :: k).
